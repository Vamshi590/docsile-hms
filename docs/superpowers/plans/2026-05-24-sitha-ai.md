import { SignJWT, jwtVerify, type JWTPayload } from "jose";

export type SessionPayload = {
  sub: string;
  email: string;
  name: string;
};

const DEFAULT_EXPIRY_SEC = 12 * 60 * 60;

function getSecret(): Uint8Array {
  const s = process.env.SESSION_JWT_SECRET;
  if (!s) throw new Error("SESSION_JWT_SECRET is not set");
  return new TextEncoder().encode(s);
}

export async function signSession(
  payload: SessionPayload,
  opts: { expiresInSec?: number } = {},
): Promise<string> {
  const exp = opts.expiresInSec ?? DEFAULT_EXPIRY_SEC;
  return await new SignJWT(payload as unknown as JWTPayload)
    .setProtectedHeader({ alg: "HS256" })
    .setIssuedAt()
    .setExpirationTime(`${exp}s`)
    .sign(getSecret());
}

export async function verifySession(token: string): Promise<SessionPayload> {
  const { payload } = await jwtVerify(token, getSecret());
  return payload as unknown as SessionPayload;
}
```

import "server-only";
import { cookies } from "next/headers";
import { cache } from "react";
import { redirect } from "next/navigation";
import { createClient } from "./supabase/server";
import { signSession, verifySession, type SessionPayload } from "./jwt";
import { verifyPassword } from "./password";
import type { DevUser } from "./supabase/types";

const SESSION_COOKIE = "sitha-ai-session";
const SESSION_MAX_AGE_SEC = 12 * 60 * 60;

export const getSession = cache(async (): Promise<SessionPayload | null> => {
  const c = await cookies();
  const token = c.get(SESSION_COOKIE)?.value;
  if (!token) return null;
  try {
    return await verifySession(token);
  } catch {
    return null;
  }
});

export async function requireSession(): Promise<SessionPayload> {
  const s = await getSession();
  if (!s) redirect("/login");
  return s;
}

export async function login(
  email: string,
  password: string,
): Promise<{ ok: true } | { ok: false; error: string }> {
  const supabase = createClient();
  const { data: user } = await supabase
    .from("DevUser")
    .select("*")
    .eq("email", email.toLowerCase())
    .maybeSingle<DevUser>();

  if (!user || !user.isActive) return { ok: false, error: "Invalid credentials" };

  const ok = await verifyPassword(password, user.passwordHash);
  if (!ok) return { ok: false, error: "Invalid credentials" };

  const token = await signSession({ sub: user.id, email: user.email, name: user.name });

  const c = await cookies();
  c.set(SESSION_COOKIE, token, {
    httpOnly: true,
    secure: process.env.NODE_ENV === "production",
    sameSite: "lax",
    path: "/",
    maxAge: SESSION_MAX_AGE_SEC,
  });

  await supabase
    .from("DevUser")
    .update({ lastLoginAt: new Date().toISOString() })
    .eq("id", user.id);

  return { ok: true };
}

export async function logout(): Promise<void> {
  const c = await cookies();
  c.delete(SESSION_COOKIE);
}
```

const PREFIX = "sta_live_";

const PREFIX_LEN = 16;

export function generateApiKey(): string {
  const body = randomBytes(KEY_BYTES).toString("base64url");
  return `${PREFIX}${body}`;
}

export function extractPrefix(key: string): string {
  return key.slice(0, PREFIX_LEN);
}

export async function hashApiKey(key: string): Promise<string> {
  return bcrypt.hash(key, COST);

- [ ] **Step 2: Write `src/lib/rate-limit.ts`**

```ts
type Bucket = { count: number; windowStart: number };

export type RateLimiter = { check: (key: string) => boolean };

export function createRateLimiter(opts: { limit: number; windowMs: number }): RateLimiter {
  const buckets = new Map<string, Bucket>();
  return {
    check(key: string): boolean {
      const now = Date.now();
      const b = buckets.get(key);
      if (!b || now - b.windowStart >= opts.windowMs) {
        buckets.set(key, { count: 1, windowStart: now });
        return true;
      }
      if (b.count >= opts.limit) return false;
      b.count += 1;
      return true;
    },
  };
}

export const publicApiLimiter = createRateLimiter({ limit: 60, windowMs: 60_000 });
```

- [ ] **Step 3: Write `src/lib/api-auth.ts`**

```ts
import "server-only";
import { createClient } from "./supabase/server";
import { verifyApiKey, extractPrefix } from "./api-keys";
import { publicApiLimiter } from "./rate-limit";
import type { Hospital } from "./supabase/types";

export type ApiAuthResult =
  | { ok: true; hospital: Hospital; keyId: string }
  | { ok: false; status: 401 | 403 | 429; code: string };

export async function authenticateRequest(
  authHeader: string | undefined,
): Promise<ApiAuthResult> {
  if (!authHeader?.startsWith("Bearer ")) {
    return { ok: false, status: 401, code: "missing_key" };
  }

  const raw = authHeader.slice(7);
  if (!publicApiLimiter.check(raw)) {
    return { ok: false, status: 429, code: "rate_limit" };
  }

  const supabase = createClient();
  const prefix = extractPrefix(raw);
  const { data: candidates } = await supabase
    .from("ApiKey")
    .select("id, hospitalId, keyHash, revokedAt")
    .eq("keyPrefix", prefix)
    .is("revokedAt", null);

  for (const cand of candidates ?? []) {
    if (await verifyApiKey(raw, cand.keyHash)) {
      const { data: hospital } = await supabase
        .from("Hospital")
        .select("*")
        .eq("id", cand.hospitalId)
        .maybeSingle();
      if (!hospital) return { ok: false, status: 401, code: "invalid_key" };

      supabase
        .from("ApiKey")
        .update({ lastUsedAt: new Date().toISOString() })
        .eq("id", cand.id)
        .then(() => {});

      return { ok: true, hospital: hospital as Hospital, keyId: cand.id };
    }
  }
  return { ok: false, status: 401, code: "invalid_key" };
}
```

export type SendEmailResult =
  | { ok: true; messageId: string }
  | { ok: false; error: string };

export async function sendEmail(input: SendEmailInput): Promise<SendEmailResult> {
  const from = process.env.SES_FROM_EMAIL;
  if (!from) return { ok: false, error: "SES_FROM_EMAIL not configured" };

  try {
    const cmd = new SendEmailCommand({
      Source: from,
      Destination: { ToAddresses: [input.to] },
      Message: {
        Subject: { Data: input.subject, Charset: "UTF-8" },
        Body: { Text: { Data: input.body, Charset: "UTF-8" } },
      },
    });
    const res = await getClient().send(cmd);
    return { ok: true, messageId: res.MessageId ?? "" };
  } catch (err) {
    return { ok: false, error: err instanceof Error ? err.message : "SES error" };
  }
}

import { NextResponse, type NextRequest } from "next/server";
import { jwtVerify } from "jose";

const SESSION_COOKIE = "sitha-ai-session";
const PUBLIC_PATHS = ["/login", "/api/v1"];

function isPublic(path: string) {
  return PUBLIC_PATHS.some((p) => path === p || path.startsWith(`${p}/`));
}

export async function proxy(req: NextRequest) {
  const { pathname } = req.nextUrl;
  if (isPublic(pathname)) return NextResponse.next();

  const token = req.cookies.get(SESSION_COOKIE)?.value;
  if (!token) {
    const url = req.nextUrl.clone();
    url.pathname = "/login";
    return NextResponse.redirect(url);
  }

  try {
    const secret = new TextEncoder().encode(process.env.SESSION_JWT_SECRET!);
    await jwtVerify(token, secret);
    return NextResponse.next();
  } catch {
    const url = req.nextUrl.clone();
    url.pathname = "/login";
    const res = NextResponse.redirect(url);
    res.cookies.delete(SESSION_COOKIE);
    return res;
  }
}

"use server";
import { z } from "zod";
import { revalidatePath } from "next/cache";
import { createClient } from "@/lib/supabase/server";
import { requireSession } from "@/lib/auth";
import { generateApiKey, hashApiKey, extractPrefix } from "@/lib/api-keys";

const CreateSchema = z.object({
  name: z.string().min(2),
  code: z.string().min(2).regex(/^[a-z0-9-]+$/, "lowercase letters, numbers and hyphens only"),
});

export type CreateHospitalResult =
  | { ok: true; hospitalId: string; apiKey: string }
  | { ok: false; error: string };

export async function createHospitalAction(formData: FormData): Promise<CreateHospitalResult> {
  await requireSession();
  const parsed = CreateSchema.safeParse({
    name: formData.get("name"),
    code: formData.get("code"),
  });
  if (!parsed.success) return { ok: false, error: parsed.error.issues[0]?.message ?? "Invalid input" };

  const supabase = createClient();

  // Check code uniqueness
  const { data: existing } = await supabase
    .from("Hospital")
    .select("id")
    .eq("code", parsed.data.code)
    .maybeSingle();
  if (existing) return { ok: false, error: "Hospital code already exists" };

  const { data: hospital, error: hErr } = await supabase
    .from("Hospital")
    .insert({ name: parsed.data.name, code: parsed.data.code })
    .select("id")
    .single();
  if (hErr || !hospital) return { ok: false, error: "Failed to create hospital" };

  const rawKey = generateApiKey();
  const { error: kErr } = await supabase.from("ApiKey").insert({
    hospitalId: hospital.id,
    keyPrefix: extractPrefix(rawKey),
    keyHash: await hashApiKey(rawKey),
    label: "default",
  });
  if (kErr) {
    await supabase.from("Hospital").delete().eq("id", hospital.id);
    return { ok: false, error: "Failed to create API key" };
  }

  revalidatePath("/hospitals");
  return { ok: true, hospitalId: hospital.id, apiKey: rawKey };
}

export async function issueNewKeyAction(
  hospitalId: string,
  label: string,
): Promise<{ ok: true; apiKey: string } | { ok: false; error: string }> {
  await requireSession();
  const rawKey = generateApiKey();
  const supabase = createClient();
  const { error } = await supabase.from("ApiKey").insert({
    hospitalId,
    keyPrefix: extractPrefix(rawKey),
    keyHash: await hashApiKey(rawKey),
    label: label || "default",
  });
  if (error) return { ok: false, error: "Failed to issue key" };
  revalidatePath("/hospitals");
  return { ok: true, apiKey: rawKey };
}

export async function revokeKeyAction(keyId: string, hospitalId: string): Promise<void> {
  await requireSession();
  const supabase = createClient();
  await supabase
    .from("ApiKey")
    .update({ revokedAt: new Date().toISOString() })
    .eq("id", keyId)
    .eq("hospitalId", hospitalId);
  revalidatePath("/hospitals");
}

export async function listHospitalsAction() {
  await requireSession();
  const supabase = createClient();
  const { data: hospitals } = await supabase
    .from("Hospital")
    .select("*")
    .order("createdAt", { ascending: false });
  return hospitals ?? [];
}

export async function listKeysAction(hospitalId: string) {
  await requireSession();
  const supabase = createClient();
  const { data: keys } = await supabase
    .from("ApiKey")
    .select("id, label, keyPrefix, lastUsedAt, revokedAt, createdAt")
    .eq("hospitalId", hospitalId)
    .order("createdAt", { ascending: false });
  return keys ?? [];
}

"use server";
import { z } from "zod";
import { revalidatePath } from "next/cache";
import { createClient } from "@/lib/supabase/server";
import { requireSession } from "@/lib/auth";

const TemplateSchema = z.object({
  code: z.string().min(2).regex(/^[a-z0-9-]+$/, "lowercase, numbers and hyphens only"),
  name: z.string().min(2),
  subjectTemplate: z.string().min(1),
  bodyTemplate: z.string().min(1),
});

type Result = { ok: true } | { ok: false; error: string };

export async function listTemplatesAction() {
  await requireSession();
  const { data } = await createClient()
    .from("EmailTemplate")
    .select("*")
    .order("createdAt", { ascending: false });
  return data ?? [];
}

export async function createTemplateAction(formData: FormData): Promise<Result> {
  await requireSession();
  const parsed = TemplateSchema.safeParse({
    code: formData.get("code"),
    name: formData.get("name"),
    subjectTemplate: formData.get("subjectTemplate"),
    bodyTemplate: formData.get("bodyTemplate"),
  });
  if (!parsed.success) return { ok: false, error: parsed.error.issues[0]?.message ?? "Invalid input" };

  const { error } = await createClient()
    .from("EmailTemplate")
    .insert({ ...parsed.data, isActive: true });
  if (error) return { ok: false, error: error.message };

  revalidatePath("/templates");
  return { ok: true };
}

export async function updateTemplateAction(id: string, formData: FormData): Promise<Result> {
  await requireSession();
  const parsed = TemplateSchema.safeParse({
    code: formData.get("code"),
    name: formData.get("name"),
    subjectTemplate: formData.get("subjectTemplate"),
    bodyTemplate: formData.get("bodyTemplate"),
  });
  if (!parsed.success) return { ok: false, error: parsed.error.issues[0]?.message ?? "Invalid input" };

  const { error } = await createClient()
    .from("EmailTemplate")
    .update(parsed.data)
    .eq("id", id);
  if (error) return { ok: false, error: error.message };

  revalidatePath("/templates");
  return { ok: true };
}

export async function toggleTemplateAction(id: string, isActive: boolean): Promise<void> {
  await requireSession();
  await createClient().from("EmailTemplate").update({ isActive }).eq("id", id);
  revalidatePath("/templates");
}

export async function deleteTemplateAction(id: string): Promise<void> {
  await requireSession();
  await createClient().from("EmailTemplate").delete().eq("id", id);
  revalidatePath("/templates");
}

import { NextRequest, NextResponse } from "next/server";
import { z } from "zod";
import { createClient } from "@/lib/supabase/server";
import { authenticateRequest } from "@/lib/api-auth";
import { renderTemplate } from "@/lib/template-renderer";
import { sendEmail } from "@/lib/ses";
import type { EmailTemplate } from "@/lib/supabase/types";

export const dynamic = "force-dynamic";

const Schema = z.object({
  templateCode: z.string().min(1),
  to: z.object({
    email: z.string().email(),
    name: z.string().default(""),
  }),
  variables: z.record(z.string(), z.string()).default({}),
});

export async function POST(req: NextRequest) {
  const auth = await authenticateRequest(req.headers.get("authorization") ?? undefined);
  if (!auth.ok) {
    return NextResponse.json({ error: auth.code }, { status: auth.status });
  }

  let body: unknown;
  try { body = await req.json(); }
  catch { return NextResponse.json({ error: "Invalid JSON" }, { status: 400 }); }

  const parsed = Schema.safeParse(body);
  if (!parsed.success) {
    return NextResponse.json({ error: parsed.error.issues[0]?.message }, { status: 400 });
  }

  const { templateCode, to, variables } = parsed.data;
  const supabase = createClient();

  const { data: template } = await supabase
    .from("EmailTemplate")
    .select("*")
    .eq("code", templateCode)
    .eq("isActive", true)
    .maybeSingle<EmailTemplate>();

  if (!template) {
    return NextResponse.json({ error: "Template not found or inactive" }, { status: 404 });
  }

  const { subject, body: emailBody } = renderTemplate(
    { subject: template.subjectTemplate, body: template.bodyTemplate },
    variables,
  );

  const result = await sendEmail({ to: to.email, toName: to.name, subject, body: emailBody });

  await supabase.from("EmailLog").insert({
    hospitalId: auth.hospital.id,
    templateCode,
    toEmail: to.email,
    toName: to.name,
    status: result.ok ? "SENT" : "FAILED",
    sesMessageId: result.ok ? result.messageId : null,
    errorMessage: result.ok ? null : result.error,
  });

  if (!result.ok) {
    return NextResponse.json({ success: false, error: result.error }, { status: 500 });
  }

  return NextResponse.json({ success: true, messageId: result.messageId });
}
```

  await requireAuth()

  const sithaUrl = process.env.SITHA_API_URL
  const sithaKey = process.env.SITHA_API_KEY

  if (!sithaUrl || !sithaKey) {
    return { ok: false, error: "Email service not configured" }
  }

  try {
    const res = await fetch(`${sithaUrl}/api/v1/email/send`, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        Authorization: `Bearer ${sithaKey}`,
      },
      body: JSON.stringify({
        templateCode: input.templateCode,
        to: { email: input.toEmail, name: input.toName },
        variables: input.variables,
      }),
    })

    const data = await res.json()
    if (!res.ok) return { ok: false, error: data.error ?? "Email failed" }
    return { ok: true }
  } catch (err) {
    return { ok: false, error: err instanceof Error ? err.message : "Network error" }
  }
}
```

## Task 18: HMS Integration — SendEmailModal

**Files:**
- Create: `src/app/(hospital)/dues-followups/components/SendEmailModal.tsx`

- [ ] **Step 1: Write `SendEmailModal.tsx`**

```tsx
"use client"

import { useState, useEffect } from "react"
import { toast } from "sonner"
import { Mail } from "lucide-react"
import { Button } from "@/components/ui/button"
import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogFooter } from "@/components/ui/dialog"
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from "@/components/ui/select"
import { Label } from "@/components/ui/label"
import { sendPatientEmail } from "../actions"
import type { FollowUpRecord } from "../actions"

type Template = { code: string; name: string }

interface Props {
  record: FollowUpRecord
  hospitalName: string
  open: boolean
  onClose: () => void
}

export function SendEmailModal({ record, hospitalName, open, onClose }: Props) {
  const [templates, setTemplates] = useState<Template[]>([])
  const [templateCode, setTemplateCode] = useState("")
  const [loading, setLoading] = useState(false)
  const [sending, setSending] = useState(false)

  useEffect(() => {
    if (!open) return
    setLoading(true)
    const sithaUrl = process.env.NEXT_PUBLIC_SITHA_API_URL
    if (!sithaUrl) { setLoading(false); return }

    fetch(`${sithaUrl}/api/v1/templates`, {
      headers: { Authorization: `Bearer ${process.env.NEXT_PUBLIC_SITHA_API_KEY ?? ""}` },
    })
      .then((r) => r.json())
      .then((data) => { setTemplates(data.templates ?? []); setLoading(false) })
      .catch(() => setLoading(false))
  }, [open])

  async function handleSend() {
    if (!templateCode) { toast.error("Select a template"); return }
    if (!record.email) { toast.error("Patient has no email address"); return }
    setSending(true)
    const result = await sendPatientEmail({
      toEmail: record.email,
      toName: record.patientName,
      templateCode,
      variables: {
        patientName: record.patientName,
        hospitalName,
        followUpDate: new Date(record.followUpDate).toLocaleDateString("en-IN", {
          day: "numeric", month: "long", year: "numeric",
        }),
        date: new Date().toLocaleDateString("en-IN", {
          day: "numeric", month: "long", year: "numeric",
        }),
      },
    })
    setSending(false)
    if (!result.ok) { toast.error(result.error); return }
    toast.success("Email sent successfully")
    onClose()
  }

  return (
    <Dialog open={open} onOpenChange={onClose}>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Send Email to {record.patientName}</DialogTitle>
        </DialogHeader>
        <div className="space-y-4 py-2">
          <div className="text-sm text-muted-foreground">
            <p>To: {record.email ?? <span className="text-destructive">No email on record</span>}</p>
          </div>
          <div className="space-y-1.5">
            <Label>Template</Label>
            {loading ? (
              <p className="text-sm text-muted-foreground">Loading templates…</p>
            ) : templates.length === 0 ? (
              <p className="text-sm text-muted-foreground">No active templates. Create one in Sitha AI.</p>
            ) : (
              <Select value={templateCode} onValueChange={setTemplateCode}>
                <SelectTrigger>
                  <SelectValue placeholder="Select a template" />
                </SelectTrigger>
                <SelectContent>
                  {templates.map((t) => (
                    <SelectItem key={t.code} value={t.code}>{t.name}</SelectItem>
                  ))}
                </SelectContent>
              </Select>
            )}
          </div>
        </div>
        <DialogFooter>
          <Button variant="outline" onClick={onClose}>Cancel</Button>
          <Button onClick={handleSend} disabled={sending || !record.email || !templateCode}>
            {sending ? "Sending…" : "Send email"}
          </Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  )
}
```

**Important:** The `SendEmailModal` fetches templates directly from Sitha AI using `NEXT_PUBLIC_SITHA_API_URL` and `NEXT_PUBLIC_SITHA_API_KEY`. Add these to HMS `.env.local`:
```bash
NEXT_PUBLIC_SITHA_API_URL=http://localhost:3001
NEXT_PUBLIC_SITHA_API_KEY=sta_live_<your_key>
```

(The `sendPatientEmail` server action uses the non-public `SITHA_API_URL`/`SITHA_API_KEY` — the public vars are only for the client-side template fetch.)
